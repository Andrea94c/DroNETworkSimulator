 44    def score(self, undelivered_penalty=1.5):
 45        """ returns a score for the exectued simulation: 
 46
 47                sum( event delays )  / number of events
 48
 49            Notice that, expired or not found events will be counted with a max_delay*penalty
 50        """
 51        # mean delivery time
 52        # get best delivery time for each event  
 53        all_delivered_events = set([pck.event_ref for pck, _ in self.drones_packets_to_depot])
 54        event_delivery_times_dict = {ev.identifier: [] for ev in all_delivered_events}
 55
 56        # DELIVERY TIME -> METRIC FOR PLOT
 57        for pck, delivery_ts in self.drones_packets_to_depot:
 58            # time between event generation and packet delivery to depot -> dict to help computation
 59            event_delivery_times_dict[pck.event_ref.identifier].append(delivery_ts - pck.event_ref.current_time)
 60
 61        # maps every event to the minimum delay of the packet arrival to the depot
 62        event_delivery_times = []
 63        for ev_id in event_delivery_times_dict.keys():
 64            event_delivery_times.append(np.nanmin(event_delivery_times_dict[ev_id]))
 65
 66        not_delivered_events = len(self.events) - len(all_delivered_events) 
 67        assert not_delivered_events >= 0
 68         
 69        event_delivery_times.extend([undelivered_penalty * self.simulator.event_duration] * not_delivered_events)  # add penalities to not delivered or not found packets
 70
 71        return np.nanmean(event_delivery_times)

 73    def other_metrics(self):
 74        """ Metrics evaluated post execution """
 75
 76        # the number of all the events generated during the simulation
 77        self.number_of_generated_events = len(self.events)
 78        self.number_of_not_generated_events = len(self.events_not_listened)
 79
 80        # the number of all events that the drones discovers, either notified or not
 81        all_detected_events = set([pck.event_ref for pck in self.drones_packets])
 82        self.number_of_detected_events = len(all_detected_events)
 83
 84        # the number of all events that the drones notify (before the event deadline) to the depot
 85        all_delivered_events = set([pck.event_ref for pck, _ in self.drones_packets_to_depot])
 86
 87        self.number_of_events_to_depot = len(all_delivered_events)
 88        self.number_of_packets_to_depot = len(self.drones_packets_to_depot)  # may contain duplicates
 89        
 90        # NOTE: THE DEPOT PACKETS ARE NOT COUNTED, WE ADD THEM HERE!! 
 91        self.all_data_packets_in_simulation += len(self.drones_packets_to_depot)
 92
 93        # mean delivery time 
 94        packet_delivery_times = []
 95
 96        # maps an event to the time steps in which it was delivered
 97        event_delivery_times_dict = {ev.identifier: [] for ev in all_delivered_events}
 98
 99        # DELIVERY TIME -> METRIC FOR PLOT
100        for pck, delivery_ts in self.drones_packets_to_depot:
101            # time between packet generation and packet delivery to depot
102            packet_delivery_times.append(delivery_ts - pck.time_step_creation)
103
104            # time between event generation and packet delivery to depot -> dict to help computation
105            event_delivery_times_dict[pck.event_ref.identifier].append(delivery_ts - pck.event_ref.current_time)
106
107        # maps every event to the minimum delay of the packet arrival to the depot
108        event_delivery_times = []
109        for ev_id in event_delivery_times_dict.keys():
110            event_delivery_times.append(np.nanmin(event_delivery_times_dict[ev_id]))
111
112        # averaged delays over all packets/events
113        self.event_delivery_times = event_delivery_times
114        self.packet_mean_delivery_time = np.mean(packet_delivery_times)
115        self.event_mean_delivery_time = np.mean(event_delivery_times)

157    def __dictionary_represenation(self):
158        """ compute the dictionary to save as json """
159        self.other_metrics()

161        out_results = {"mission_setup": self.mission_setup}

167        out_results["number_of_packets_to_depot"] = self.number_of_packets_to_depot
168        out_results["packet_mean_delivery_time"] = self.packet_mean_delivery_time
169        out_results["event_mean_delivery_time"] = self.event_mean_delivery_time
170        out_results["time_on_mission"] = self.time_on_mission
171        out_results["all_control_packets_in_simulation"] = self.all_control_packets_in_simulation
172        out_results["all_data_packets_in_simulation"] = self.all_data_packets_in_simulation
173        out_results["all_events"] = [ev.to_json() for ev in self.events]
174        out_results["not_listened_events"] = [ev.to_json() for ev in self.events_not_listened]
175        out_results["events_delivery_times"] = [str(e) for e in self.event_delivery_times]
176        out_results["drones_packets"] = [pck.to_json() for pck in self.drones_packets]
177        out_results["drones_to_depot_packets"] = [(pck.to_json(), delivery_ts) for pck, delivery_ts in self.drones_packets_to_depot]
178        out_results["score"] = self.score()
179        out_results["mean_number_of_relays"] = np.nanmean(self.mean_numbers_of_possible_relays)
180
181        return out_results
