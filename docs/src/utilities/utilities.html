 16def compute_circle_path(radius : int, center : tuple) -> list:
 17    """ compute a set of finite coordinates to simulate a circle trajectory of input radius around a given center
 18
 19        radius : int -> the radius of the trajectory
 20        centers : tuple (x, y) the center of the trajectory
 21        return a list of tuple (coordinates)
 22    """
 23    x = list(range(-radius, radius))
 24    coords = []
 25    for x_ in x:
 26        y_ = radius ** 2 - (x_) ** 2
 27        coords.append((x_, (y_ ** (0.5))))
 28    coords2 = coords[::-1]
 29    coords2 = [(x, -y) for x,y in coords2]
 30    coords += coords2
 31    return [(x + center[0], y + center[1]) for x,y in coords]

 37def euclidean_distance(p1, p2):
 38    """ Given points p1, p2 in R^2 it returns the norm of the vector connecting them.  """
 39    dist = ((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2) ** 0.5
 40    return dist

 73    def handle_events_generation(self, cur_step : int, drones : list):
 74        """
 75        at fixed time randomly select a drone from the list and sample on it a packet/event.
 76
 77        :param cur_step: the current step of the simulation to decide whenever sample an event or not
 78        :param drones: the drones where to sample the event
 79        :return: nothing
 80        """
 81        if cur_step % self.simulator.event_generation_delay == 0:  # if it's time to generate a new packet
 82            # drone that will receive the packet:
 83            drone_index = self.rnd_drones.randint(0, len(drones))
 84            drone = drones[drone_index]
 85            drone.feel_event(cur_step)

128    def __cirlce_path(self, drone_id, simulator, center=None, radius=None):
129        if center is None:

150def json_to_paths(json_file_path):
151    """ load the tour for drones
152        and return a dictionary {drone_id : list of waypoint}
153
154        e.g.,
155        accept json that contains:
156        {"drones": [{"index": "0", "tour": ["(1500, 0)", "(1637, 172)", ...
157                    (1500, 0)"]}, {"index": "1", "tour": ["(1500, 0)",
158
159        TOURS = {
160            0 : [(0,0), (2000,2000), (1500, 1500), (200, 2000)],
161            1 : [(0,0), (2000, 200), (200, 2000), (1500, 1500)]
162        }
163    """
164    out_data = {}
165    with open(json_file_path, 'r') as in_file:
166        data = json.load(in_file)
167        for drone_data in data["drones"]:
168            drone_index = int(drone_data["index"])
169            drone_path = []
170            for waypoint in drone_data["tour"]:
171                drone_path.append(make_tuple(waypoint))
172            out_data[drone_index] = drone_path
173    return out_data

18def compute_circle_path(radius : int, center : tuple) -> list:
19    """ compute a set of finite coordinates to simulate a circle trajectory of input radius around a given center
20
21        radius : int -> the radius of the trajectory
22        centers : tuple (x, y) the center of the trajectory
23        return a list of tuple (coordinates)
24    """
25    x = list(range(-radius, radius))
26    coords = []
27    for x_ in x:
28        y_ = radius ** 2 - (x_) ** 2
29        coords.append((x_, (y_ ** (0.5))))
30    coords2 = coords[::-1]
31    coords2 = [(x, -y) for x,y in coords2]
32    coords += coords2
33    return [(x + center[0], y + center[1]) for x,y in coords]

277        min_y, max_y = min(coords_cell_start[1], coords_cell_end[1]), max(coords_cell_start[1], coords_cell_end[1])
278
279        # All the cells of the rectangle, indices

@staticmethod
def intersect_segments(start1: tuple, end1: tuple, start2: tuple, end2: tuple): 
301    @staticmethod
302    def intersect_segments(start1:tuple, end1:tuple, start2:tuple, end2:tuple):
303        if end1 == start2:
304            return True
305        if end2 == start1:
306            return True
307        if start2 == start1:
308            return True
309        if end2 == end1:
310            return True
311
312        a = np.asarray(end1) - np.asarray(start1)  # direction of line a
313        b = np.asarray(start2) - np.asarray(end2)  # direction of line b, reversed
314        d = np.asarray(start2) - np.asarray(start1)  # right-hand side
315        det = a[0] * b[1] - a[1] * b[0]
316
317        if det == 0:
318            return False
319
320        t = (a[0] * d[1] - a[1] * d[0]) / det
321        return 0 <= t <= 1
